{% macro format_var(input_string) -%}
    {%- if input_string -%}
        {#- Список "неправильных" символов, которые нужно удалить с начала строки -#}
        {%- set chars_to_strip = '$@#%' -%}
        {#- Убираем все неправильные символы с начала и добавляем один $ впереди -#}
        {{- '$' ~ input_string.lstrip(chars_to_strip) -}}
    {%- endif -%}
{%- endmacro %}

{% macro has_attribute(element, attribute_name) %}
    {% set class_name = CLASS_NAME_MAP.get(element.elem_class) %}
    {% if class_name and attribute_name in CLASS_ATTRIBUTES.get(class_name, []) %}
        true
    {% endif %}
{% endmacro %}

{% macro anchors_generate(mod_file) %}
{% if mod_file.scene_data.rzm and mod_file.scene_data.rzm.elements %}
    {% for element in mod_file.scene_data.rzm.elements %}
        {% if has_attribute(element, 'isMovable') %}
        global persist ${{element.element_name}}PositionX = {{ (element.position | default([0,0]))[0] }}
        global persist ${{element.element_name}}PositionY = {{ (element.position | default([0,0]))[1] }}
        {% endif %}
        {% if has_attribute(element, 'isResizable') %}
        {% if CLASS_NAME_MAP[element.elem_class] == 'GRID_CONTAINER'%}
        global persist ${{element.element_name}}scroll = 0
        global persist ${{element.element_name}}PositionX = {{ (element.position | default([0,0]))[0] }}
        global persist ${{element.element_name}}PositionY = {{ (element.position | default([0,0]))[1] }}
        {% endif%}
        global persist ${{element.element_name}}SizeX = {{ (element.size | default([0,0]))[0] }}
        global persist ${{element.element_name}}SizeY = {{ (element.size | default([0,0]))[1] }}
        {% endif %}
        {% if CLASS_NAME_MAP[element.elem_class] == 'ANCHOR'%}
        global persist ${{element.element_name}}SizeX = {{ (element.size | default([0,0]))[0] }}
        global persist ${{element.element_name}}SizeY = {{ (element.size | default([0,0]))[1] }}
        {% endif %}
    {% endfor %}
{% endif %}
{% endmacro %}

{% macro customDrawVars(mod_file) %}
    {% for component in mod_file.components if component.draw_vb != "" %}
        {% for part in component.parts if part.vertex_count > 0 %}
            {% for entry in part.objects if entry.vertex_count > 0 %}
                {% for key, value in entry.obj.items() %}
                    {% if key.startswith('CustomDraw') %}
                        global $CDC_{{ key|replace('CustomDraw', '') }} = {{ entry.index_count }}
                        global $CDO_{{ key|replace('CustomDraw', '') }} = {{ entry.index_offset }}
                    {% endif %}
                {% endfor %}
            {% endfor %}
        {% endfor %}
    {% endfor %}
{%endmacro%}

{% macro shape_generate(mod_file) %}
