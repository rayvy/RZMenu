[CommandListCoreFNComputeMouseInteractions.Move]
local $loordsX
local $loordsY

if $capturedID == $id
    if $dragState == 0
			$dragState = 1
			$loordsX = $cursorX - $positionX
			$loordsY = $cursorY - $positionY
	else if $dragState == 1
		$positionX = $cursorX - $loordsX
		$positionY = $cursorY - $loordsY
	endif
endif

[CommandListCoreFNComputeMouseInteractions.GridScroll]
local $loordsX
local $loordsY

if $capturedID == $id
	if $dragState == 0
		$dragState = 1
		$loordsX = $cursorX - $positionX
	else if $dragState == 1
		$RZM_OUT_VAL = ($cursorX - $loordsX * 0.1)
	endif
endif


[CommandListCoreFNComputeMouseInteractions.Resize]
local $local_ResizeOffsetX
local $local_ResizeOffsetY

if $capturedID == $id
	if $dragState == 0
		if $cursorX > ($positionX+$sizeX*0.9) && $cursorX < ($positionX+$sizeX) && $cursorY > ($positionY+$sizeY*0.9) && $cursorY < ($positionY+$sizeY)
			$dragState = 2
			$local_ResizeOffsetX = $cursorX - ($positionX + $sizeX)
			$local_ResizeOffsetY = $cursorY - ($positionY + $sizeY)
		endif
	else if $dragState == 2
		$sizeX = ($cursorX - $local_ResizeOffsetX) - $positionX
		$sizeY = ($cursorY - $local_ResizeOffsetY) - $positionY
	endif
endif

[CommandListCoreFNHandleHold]
local $TimeHoldCounter

if $capturedID == $id
    if $holdTimer > $holdThreshold && $dragState == 0
        $dragState = 3
        $TimeHoldCounter = $holdTimer
    endif
    if $dragState == 3
    $TimeHoldCounter = $TimeHoldCounter + $dt
    w33 = $holdTimer
    endif
endif
